# qb-builderjob
Builder Job For QB-Core

# How to install

Add the builder job to your shared.lua

["builder"] = {
        label = "Builder",
	defaultDuty = true,
	grades = {
            ['0'] = {
                name = "Recruit",
                payment = 25
            },
        },
},

Add to the city hall
